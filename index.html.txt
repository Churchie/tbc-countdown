  <script>
    // === EDIT THIS IF YOU MEANT A DIFFERENT YEAR ===
    const TARGET_YEAR = 2026;
    // Target: February 5 at 23:00 GMT (UTC)
    const targetUtcMs = Date.UTC(TARGET_YEAR, 1, 5, 23, 0, 0); // month is 0-based (1 = Feb)
    const targetDate = new Date(targetUtcMs);

    const el = (id) => document.getElementById(id);
    const pad2 = (n) => String(n).padStart(2, "0");

    el("targetText").textContent = targetDate.toISOString().replace(".000Z","Z");

    function tick(){
      const now = Date.now();
      let diff = targetUtcMs - now;

      if (diff <= 0){
        el("days").textContent = "00";
        el("hours").textContent = "00";
        el("minutes").textContent = "00";
        el("seconds").textContent = "00";
        el("status").innerHTML = 'Status: <span class="done">The Dark Portal is OPEN.</span>';
        return;
      }

      const totalSeconds = Math.floor(diff / 1000);

      const days = Math.floor(totalSeconds / 86400);
      const hours = Math.floor((totalSeconds % 86400) / 3600);
      const minutes = Math.floor((totalSeconds % 3600) / 60);
      const seconds = totalSeconds % 60;

      el("days").textContent = String(days);
      el("hours").textContent = pad2(hours);
      el("minutes").textContent = pad2(minutes);
      el("seconds").textContent = pad2(seconds);

      el("localText").textContent = new Date(targetUtcMs).toLocaleString(undefined, {
        year: "numeric", month: "short", day: "2-digit",
        hour: "2-digit", minute: "2-digit", second: "2-digit"
      });
    }

    tick();
    setInterval(tick, 250);
  </script>
